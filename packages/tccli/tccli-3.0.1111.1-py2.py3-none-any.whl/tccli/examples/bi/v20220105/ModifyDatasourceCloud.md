```
tccli bi ModifyDatasourceCloud --cli-unfold-argument  \
    --ServiceType abc \
    --DbType abc \
    --Charset abc \
    --DbUser abc \
    --DbPwd abc \
    --DbName abc \
    --SourceName abc \
    --ProjectId abc \
    --Vip abc \
    --Vport abc \
    --VpcId abc \
    --UniqVpcId abc \
    --RegionId abc \
    --ExtraParam abc \

**Example 2: 更新云数据库示例**

```
tccli bi ModifyDatasourceCloud --cli-unfold-argument  \
    --Id 1 \
    --ProjectId abc \
    --SourceName 数据库别名 \
    --Vport 3306 \
    --VpcId 123456 \
    --Charset utf8 \
    --ExtraParam  \
    --DbUser root \

{
